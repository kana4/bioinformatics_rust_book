# Add, Subtract, Multiply, Divide

```
fn main(){

    // Make a variable a that holds 1
    let a = 1;

    // Make a variable b that holds 2
    let b = 2;

    // Add them together to get c
    let c = a + b;

    // Multiple by 3!
    let d = c * 3;
    
    // Divide by 3!
    let e = d / 3;

    // The variable e holds 3
}
```
